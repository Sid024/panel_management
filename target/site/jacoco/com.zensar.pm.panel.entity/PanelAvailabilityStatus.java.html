package com.zensar.pm.panel.entity;

import java.time.LocalDate;


import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.Id;


import lombok.Data;


@Data

@Entity
public class PanelAvailabilityStatus {

	

	@Column(name = "Email")
	private String email;

	private LocalDate toDate;
	private LocalDate fromDate;
	
	@Id
	@Column(name = "Panel Id")
	private int panelId;

	@Column(name = "Panal name")
	private String panelName;

	@Column(name = "Role")
	private String panalRole;

	@Column(name = "Interview Type")
	private String interviewType;

	@Column(name = "Availability Status")
	private String availabilityStatus;

	

	public PanelAvailabilityStatus(String email, LocalDate toDate, LocalDate fromDate, int panelId, String panelName,
			String panalRole, String interviewType, String availabilityStatus) {
		super();
		this.email = email;
		this.toDate = toDate;
		this.fromDate = fromDate;
		this.panelId = panelId;
		this.panelName = panelName;
		this.panalRole = panalRole;
		this.interviewType = interviewType;
		this.availabilityStatus = availabilityStatus;
	}



	public PanelAvailabilityStatus() {
		super();
	}

	
	
	
	

}
